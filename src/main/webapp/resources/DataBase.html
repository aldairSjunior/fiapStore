create table TB_PRODUTO ( COD_PRODUTO NUMBER not null primary key, NOME_PRODUTO VARCHAR2(100) not null, VALOR_PRODUTO NUMBER(10, 2) not null, DATA_FABRICACAO DATE, QTDE_PRODUTO NUMBER, COD_CATEGORIA NUMBER constraint COD_CATEGORIA references TB_CATEGORIA ) create table TB_CATEGORIA ( COD_CATEGORIA NUMBER not null primary key, NOME_CATEGORIA VARCHAR2(50) not null ) create table TB_USUARIO ( DS_EMAIL VARCHAR2(150) not null primary key, DS_SENHA VARCHAR2(50) not null ) CREATE SEQUENCE SQ_TB_PRODUTO MINVALUE 1 START WITH 1 INCREMENT BY 1; CREATE SEQUENCE SQ_TB_CATEGORIA MINVALUE 1 START WITH 1 INCREMENT BY 1;
